\smalltitle{سوال 1}
\begin{enumerate}
    \item \phantom{text}\\
    \begin{latin}
        \begin{center}
            
\begin{equation*}
            \begin{matrix}
                    & x_2 & y_2 \\
                   x_1 & \circs{3},\circs{7} & 6,6 \\
                   y_1 & 2,2 & \circs{7},\circs{3} 
                \end{matrix}
            \end{equation*}
       \end{center}
       (x1,x2) and (y1,y2) are the nash equilibriums.
    \end{latin}
    \item \phantom{text}\\
    \begin{latin}
        \begin{center}
            
\begin{equation*}
            \begin{matrix}
                    & \cancel{x_2} & y_2 \\
                   x_1 & \circs{7},\cancel{3} & \circs{6} ,\circs{6}  \\
                   \cancel{y_1} & \cancel{2},\cancel{2} & \cancel{3},\circs{7}  
                \end{matrix}
            \end{equation*}
       \end{center}
       (x1,y2) is the nash equilibrium.
    \end{latin}
    \item \phantom{text}\\
    \begin{latin}
        \begin{center}
            
\begin{equation*}
            \begin{matrix}
                    & x_2 & y_2 & z_2 \\
                   x_1 & 0,4 & 5,6 & \circs{8},\circs{7} \\
                   y_1 & \circs{2},\circs{9} & \circs{6},5 & 5,1   
                \end{matrix}
            \end{equation*}
       \end{center}
       (x2,y1) and (x1,z2) are the nash equilibriums.
    \end{latin}
    \item \phantom{text}\\
    \begin{latin}
        \begin{center}
            
\begin{equation*}
            \begin{matrix}
                & x_2 & y_2 & z_2 \\
                x_1 & 0,0 & \circs{5},4 & 4,\circs{5} \\
                y_1 & 4,\circs{5} & 0,0 & \circs{5},4 \\
                z_1 & \circs{5},4 & 4,\circs{5} & 0,0 
                \end{matrix}
            \end{equation*}
       \end{center}
       There is no pure nash equilibrium here.
    \end{latin}
\end{enumerate}
