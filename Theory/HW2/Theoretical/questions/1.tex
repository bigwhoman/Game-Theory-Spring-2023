        $E(U_2((p_1,p_2),R,(r_1,r_2))) = -4 * p_1 * r_1 + 0 * p_2 * r_1 + 1 * p_1 * r_2 + 2 * p_2 * r_2$\\

            $E(U_1(T,...,...)) = E(U_1(B,...,...))$\\

            $E(U_3(...,...,X)) = E(U_3(...,...,Y))$\\

    $ \begin{cases}
        q > \frac{1}{4} & Y,Z \rightarrow \text{X is dominated}\\
        q \leq \frac{1}{4} & X \rightarrow \text{Y,Z are dominated}

                            \end{equation*}
                       \end{center}
